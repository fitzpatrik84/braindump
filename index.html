const SHEET_NAME = 'Braindump';
const ALLOWED_EMAIL = '[email protected]'; // ← Byt ut mot din e-post

function doPost(e) {
  if (!checkAuth()) return deny();
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName(SHEET_NAME);
  if (!sheet) {
    sheet = ss.insertSheet(SHEET_NAME);
    sheet.appendRow(['id','text','kategori','tags','sammanfattning','actions','keywords','createdAt','date']);
    sheet.setFrozenRows(1);
  }
  const d = JSON.parse(e.postData.contents);
  if (d.action === 'save') {
    sheet.appendRow([d.id,d.text,d.kategori,d.tags,d.sammanfattning,d.actions,d.keywords,d.createdAt,d.date]);
    return ok({saved:true});
  }
  if (d.action === 'update') {
    const data = sheet.getDataRange().getValues();
    for (let i = 1; i < data.length; i++) {
      if (data[i][0] == d.id) { sheet.getRange(i+1,6).setValue(d.actions); return ok({updated:true}); }
    }
  }
  if (d.action === 'delete') {
    const data = sheet.getDataRange().getValues();
    for (let i = 1; i < data.length; i++) {
      if (data[i][0] == d.id) { sheet.deleteRow(i+1); return ok({deleted:true}); }
    }
  }
  return ok({error:'unknown action'});
}

function doGet(e) {
  if (!checkAuth()) return deny();
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName(SHEET_NAME);
  if (!sheet) return ok([]);
  const data = sheet.getDataRange().getValues();
  const rows = data.slice(1).map(r => ({
    id:r[0],text:r[1],kategori:r[2],tags:r[3],sammanfattning:r[4],
    actions:r[5],kopplingarNyckelord:r[6],createdAt:r[7],date:r[8]
  }));
  return ok(rows.reverse());
}

function checkAuth() {
  try {
    const email = Session.getActiveUser().getEmail();
    return email === ALLOWED_EMAIL;
  } catch(e) { return false; }
}

function ok(data) {
  return ContentService.createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}

function deny() {
  return ContentService.createTextOutput(JSON.stringify({error:'unauthorized'}))
    .setMimeType(ContentService.MimeType.JSON);
}
⚠ Byt ut [email protected] mot din egna Google-adress på rad 2.
3
Publicera som webbapp — kör som dig
I Apps Script: Distribuera → Ny distribution → Webbapp.

Viktigt att sätta:
· Kör som: Mig (din e-post)
· Vem har åtkomst: Alla

Klicka Distribuera, godkänn behörigheter och kopiera URL:en.
✓ Säkerheten sköts av checkAuth() i koden — inte av Google-åtkomstinställningen. Därför kan åtkomst stå på "Alla" medan din e-post fortfarande är den enda som godkänns.
4
Klistra in din webbapp-URL och e-post
Kontrollera att URL:en slutar på /exec och att e-posten är giltig.
